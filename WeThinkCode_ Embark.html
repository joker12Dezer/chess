def format_phone_number(number: str):
str_number=str(number)
formated="+27"
for n in range(1,len(str_number)):
if n==1:
formated+="("
if n==3:
formated+=")"
if (n)%3==0 and (n)!=9 :
formated+=" "
formated+=str_number[n]
return formated

def atbash(text):
alpha_low=[chr(i) for i in range(ord("z"),ord("a")+1,-1)]
alpha_upper=[chr(i) for i in range(ord("A"),ord("Z"),-1)]
ciphered=""
for i in range(len(text)):
if text[i].isupper():
ciphered+=alpha_upper[ord(text[i])-64]
elif text[i].islower():
ciphered+=alpha_low[ord(text[i])-97]
else:
ciphered+=text[i]
print(ciphered)
